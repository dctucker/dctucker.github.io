75 inline void convertOneData(int32_t* value) {
76 // Floating point
77 if (rawDataFormat == RAW_DATA_FLOAT) convertFloatToIntAtMemoryLocation((uint32_t*)value);
78
79 // Or endianness swap
80 else if (rawDataFormat == RAW_DATA_ENDIANNESS_WRONG_32) {
81 *value = swapEndianness32(*value);
82 }
83
84 else if (rawDataFormat == RAW_DATA_ENDIANNESS_WRONG_16) {
85 *value = swapEndianness2x16(*value);
86 }
87
88 else if (rawDataFormat == RAW_DATA_UNSIGNED_8) {
89 *value ^= 0x80808080;
90 }
91 }
