33 int32_t a = multiply_32x32_rshift32_rounded(input - memory, moveability) << 1;
34 int32_t b = a + memory;
35 memory = b + a;
36 return b;
37 }

39 inline int32_t doAPF(int32_t input, int32_t moveability) {
40 int32_t a = multiply_32x32_rshift32_rounded(input - memory, moveability) << 1;
