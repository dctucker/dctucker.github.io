107inline int32_t getTotalDifferenceAbs(int32_t* totals1, int32_t* totals2) {
108 int32_t totalDifferenceAbs = 0;
109 for (int i = 0; i < TIME_STRETCH_CROSSFADE_NUM_MOVING_AVERAGES; i++) {
110 int32_t differenceAbsHere = totals2[i] - totals1[i];
111 if (differenceAbsHere < 0) differenceAbsHere = -differenceAbsHere;
112 totalDifferenceAbs += differenceAbsHere;
113 }
114 return totalDifferenceAbs;
115}
116
117inline int32_t getTotalChange(int32_t* totals1, int32_t* totals2) {
118 int32_t totalChange = 0;
119 for (int i = 0; i < TIME_STRETCH_CROSSFADE_NUM_MOVING_AVERAGES; i++) {
120 totalChange += totals2[i];
121 }
122
123 for (int i = 0; i < TIME_STRETCH_CROSSFADE_NUM_MOVING_AVERAGES; i++) {
124 totalChange -= totals1[i];
125 }
126 return totalChange;
127}
