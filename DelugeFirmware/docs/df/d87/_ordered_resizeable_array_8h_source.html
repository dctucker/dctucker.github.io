51 inline int32_t getKeyAtMemoryLocation(void* address) {
52 int32_t keyBig = *(uint32_t*)((uint32_t)address + keyOffset) << keyShiftAmount;
53 return keyBig >> keyShiftAmount; // We use shifting instead of a mask so negative numbers get treated directly.
54 }
55
56 inline void setKeyAtMemoryLocation(int32_t key, void* address) {
57 uint32_t offsetAddress = (uint32_t)address + keyOffset;
58 uint32_t prevContents = *(uint32_t*)offsetAddress;
59 *(uint32_t*)offsetAddress = (key & keyMask) | (prevContents & ~keyMask);
60 }
61
62private:
63 const uint32_t keyMask;
64 const int keyOffset;
65 const int keyShiftAmount;
66};
