90 inline void writeResampled(int32_t toDelayL, int32_t toDelayR, int32_t strength1, int32_t strength2, DelayBufferSetup* setup) {
91 // If delay buffer spinning above sample rate...
92 if (setup->actualSpinRate >= 16777216) {
93
94 // An improvement on that could be to only do the triangle-widening when we're down near the native rate - i.e. set a minimum width of double the native rate rather than
95 // always doubling the width. The difficulty would be ensuring that we compensate perfectly the strength of each write so that the volume remains the same. It could totally be done.
96 // The only real advantage would be that the number of memory writes would be halved at high speeds.
97
98 // For efficiency, we start far-right, then traverse to far-left.
99 int32_t howFarRightToStart = (strength2 + (setup->spinRateForSpedUpWriting >> 8)) >> 16; // I rearranged some algebra to get this from the strengthThisWrite equation
100 int32_t distanceFromMainWrite = (int32_t)howFarRightToStart << 16; // This variable represents one "step" of the delay buffer as 65536. Always positive - absolute distance
101
102 StereoSample* writePos = bufferCurrentPos - delaySpaceBetweenReadAndWrite + howFarRightToStart; // Initially is the far-right right pos, not the central "main" one
103 while (writePos < bufferStart) writePos += sizeIncludingExtra;
104 while (writePos >= bufferEnd) writePos -= sizeIncludingExtra;
105
106 // Do all writes to the right of the main write pos
107 while (distanceFromMainWrite != 0) { // For as long as we haven't reached the "main" pos...
108 int32_t strengthThisWrite = (0xFFFFFFFF >> 4) - (((distanceFromMainWrite - strength2) >> 4) * setup->divideByRate); // Check my notebook for a rudimentary diagram
109
110 writePos->l += multiply_32x32_rshift32(toDelayL, strengthThisWrite) << 3;
111 writePos->r += multiply_32x32_rshift32(toDelayR, strengthThisWrite) << 3;
112
113 if (--writePos == bufferStart - 1) writePos = bufferEnd - 1;
114 //writePos = (writePos == bufferStart) ? bufferEnd - 1 : writePos - 1;
115 distanceFromMainWrite -= 65536;
116 }
117
118 // Do all writes to the left of (and including) the main write pos
119 while (true) {
120 int32_t strengthThisWrite = (0xFFFFFFFF >> 4) - (((distanceFromMainWrite + strength2) >> 4) * setup->divideByRate);
121 if (strengthThisWrite <= 0) break; // And stop when we've got far enough left that we shouldn't be squirting any more juice here
122
123 writePos->l += multiply_32x32_rshift32(toDelayL, strengthThisWrite) << 3;
124 writePos->r += multiply_32x32_rshift32(toDelayR, strengthThisWrite) << 3;
125
126 if (--writePos == bufferStart - 1) writePos = bufferEnd - 1;
127 distanceFromMainWrite += 65536;
128 }
129 }
130
131
132 // Or if delay buffer spinning below sample rate...
133 else {
134 // The most basic version of this would be to write to the "main" pos with strength1 and the "main + 1" pos with strength 2.
135 // But this isn't immune to aliasing, so we instead "squirt" things a little bit wider - wide enough that our "triangle" is always
136 // at least as wide as 1 step / sample in the delay buffer.
137 // This means we potentially need a further 1 write in each direction.
138
139 // And because we're "arbitrarily" increasing the width and height (height is more a side-effect of the simple algorithm) of our squirt,
140 // and also how spaced out the squirts are, the value written at each step needs to be resized.
141 // See delayWriteSizeAdjustment, which is calculated above.
142
143 // We've also had to make sure that the "triangles"' corners exactly meet up. Unfortunately this means even a tiny slow-down causes half the bandwidth to be lost
144
145 StereoSample* writePos = bufferCurrentPos - delaySpaceBetweenReadAndWrite + 2; // The furthest right we need to write is 2 steps right from the "main" write
146 while (writePos < bufferStart) writePos += sizeIncludingExtra;
147 //while (writePos >= bufferEnd) writePos -= sizeIncludingExtra; // Not needed - but be careful! Leave this here as a reminder
148
149 int32_t strength[4];
150
151 strength[1] = strength1 + setup->rateMultiple - 65536; // For the "main" pos
152 strength[2] = strength2 + setup->rateMultiple - 65536; // For the "main + 1" pos
153
154 // Strengths for the further 1 write in each direction
155 strength[0] = strength[1] - 65536;
156 strength[3] = strength[2] - 65536;
157
158 int8_t i = 3;
159 while(true) {
160 if (strength[i] > 0) {
161 writePos->l += multiply_32x32_rshift32(toDelayL, (strength[i] >> 2) * setup->writeSizeAdjustment) << 2;
162 writePos->r += multiply_32x32_rshift32(toDelayR, (strength[i] >> 2) * setup->writeSizeAdjustment) << 2;
163 }
164 if (--i < 0) break;
165 if (--writePos == bufferStart - 1) writePos = bufferEnd - 1;
166 }
167 }
168 }
