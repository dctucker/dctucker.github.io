45 inline bool equals(String* otherString) {
46 if (stringMemory == otherString->stringMemory) return true; // Works if both lengths are 0, too
47 if (!stringMemory || !otherString->stringMemory) return false; // If just one is empty, then not equal
48 return equals(otherString->get());

51 inline bool equalsCaseIrrespective(String* otherString) {
52 if (stringMemory == otherString->stringMemory) return true; // Works if both lengths are 0, too
